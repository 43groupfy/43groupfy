### Pig Code

```python
class Solution:
    def binaryBits(self, number:int) -> int:
        count = 0
        while number:
            number = number // 2
            count += 1
        return count - 1

    def rangeBitwiseAnd(self, left: int, right: int) -> int:
        if left == right:
            return left
        leftbits = self.binaryBits(left)
        rightbits = self.binaryBits(right)
        if leftbits == rightbits and leftbits >= 0:
            return 2**leftbits + self.rangeBitwiseAnd(left-2**leftbits, right-2**rightbits)
        else:
            return 0
```

### Good Code(Python)

```python
class Solution:
    def rangeBitwiseAnd(self, left: int, right: int) -> int:
        if len(bin(left)) != len(bin(right)):
            return 0
        shift = 0
        while left != right:
            left = left >> 1
            right = right >> 1
            shift += 1
        return left << shift
```
