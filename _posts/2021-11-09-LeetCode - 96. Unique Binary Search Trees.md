### Pig Code

```python
class Solution:
    def numTrees(self, n: int) -> int:
        if n == 1 or n == 0:
            return 1
        memoize = [None for i in range(n)]
        counter = 0
        for i in range(n):
            if not memoize[i]:
                memoize[i] = self.numTrees(i)
            if not memoize[n-i-1]:
                memoize[n-i-1] = self.numTrees(n-i-1)
            left = memoize[i]
            right = memoize[n-i-1]
            counter += left*right
            print(i, counter)
        return counter
```

### Good Code(Python)
```python
class Solution:
    @cache
    def numTrees(self, n: int) -> int:
        if n <= 1: return 1
        return sum(self.numTrees(i-1) * self.numTrees(n-i) for i in range(1, n+1))
```
