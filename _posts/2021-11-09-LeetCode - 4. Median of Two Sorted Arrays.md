class Solution:
    def findMedianSortedArrays(self, nums1: List[int], nums2: List[int]) -> float:
        if len(nums1) > len(nums2):
            nums2, nums1 = nums1, nums2
        m = len(nums1)
        n = len(nums2)
        total = m+n
        short_pointer = m//2-1
        long_pointer = total//2-short_pointer-2
        while True:
            short_low = float('-inf') if short_pointer < 0 else nums1[short_pointer]
            long_low = float('-inf') if long_pointer < 0 else nums2[long_pointer]
            short_high = float('inf') if short_pointer >= m-1 else nums1[short_pointer+1]
            long_high = float('inf') if long_pointer >= n-1 else nums2[long_pointer+1]
            if short_low > long_high:
                short_pointer -= 1
                long_pointer += 1
            elif long_low > short_high:
                short_pointer += 1
                long_pointer -= 1
            else:
                if total % 2 == 0:
                    return (max(short_low, long_low) + min(short_high, long_high))/2
                else:
                    return min(short_high, long_high)
